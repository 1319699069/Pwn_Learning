# 安装 pwntools

 pip install pwntools

# GDB 插件安装

git clone https://github.com/pwndbg/pwndbg //发现没有安装git

sudo apt install git //发现安装不了git，百度了一下安装方法

apt-get update -y

apt-get upgrade -y//运行，确保系统和apt包列表完全更新

//重新安装git，成功

//重新用git安装pwngdb，成功

cd pwndbg//进入pwndbg文件夹

git init//初始化

./setup.sh//运行，安装
